class RankingGameSession extends _$RankingGameSession {
  Timer? _gameTimer;

  @override
  RankingGameSessionState build(String difficulty) {
    ref.onDispose(() => _gameTimer?.cancel());
    return RankingGameSessionState.initial(difficulty);
  }

  Future<void> startGame() async {
    // å˜èªã‚’ãƒ­ãƒ¼ãƒ‰
    final words = await ref.read(
      rankingGameWordsProvider(state.difficulty).future
    );

    state = state.copyWith(
      wordQueue: words..shuffle(),
      currentWord: words.first,
      isPlaying: true,
    );

    // ã‚¿ã‚¤ãƒãƒ¼é–‹å§‹
    _gameTimer = Timer.periodic(const Duration(milliseconds: 100), (_) {
      _tick();
    });
  }

  void _tick() {
    if (state.remainingTimeMs <= 0) {
      _endGame();
      return;
    }

    state = state.copyWith(
      remainingTimeMs: state.remainingTimeMs - 100,
    );
  }

  void processInput(String key) {
    if (!state.isPlaying || state.currentWord == null) return;

    final newBuffer = state.inputBuffer + key;
    final target = state.currentWord!.word;

    // æ­£è§£ãƒã‚§ãƒƒã‚¯
    if (newBuffer == target) {
      _onCorrectAnswer();
    } else if (!target.startsWith(newBuffer)) {
      // å…¥åŠ›ãƒŸã‚¹
      _onMistake();
    } else {
      // å…¥åŠ›ç¶™ç¶šï¼ˆæ­£ã—ã„ãŒæœªå®Œäº†ï¼‰
      state = state.copyWith(
        inputBuffer: newBuffer,
        comboMeter: state.comboMeter.onCorrectKey(),
      );
      _checkBonusTime();
    }
  }

  void _onCorrectAnswer() {
    final word = state.currentWord!;
    final newCombo = state.currentCombo + 1;
    final scoreGain = _calculateScore(word.charCount, newCombo);
    final newScore = state.score + scoreGain;

    // æ¬¡ã®å˜èªã‚’å–å¾—
    final nextWordIndex = state.wordQueue.indexOf(word) + 1;
    final nextWord = nextWordIndex < state.wordQueue.length
        ? state.wordQueue[nextWordIndex]
        : state.wordQueue[0]; // ãƒ«ãƒ¼ãƒ—

    state = state.copyWith(
      score: newScore,
      correctCount: state.correctCount + 1,
      currentCombo: newCombo,
      maxCombo: max(state.maxCombo, newCombo),
      characterLevel: _calculateCharacterLevel(newScore),
      currentWord: nextWord,
      inputBuffer: '',
    );
  }

  void _onMistake() {
    state = state.copyWith(
      currentCombo: 0,
      inputBuffer: '',
      comboMeter: state.comboMeter.onMistake(),
    );
  }

  void _checkBonusTime() {
    final bonusTime = state.comboMeter.consumeBonusTime();
    if (bonusTime > 0) {
      state = state.copyWith(
        remainingTimeMs: state.remainingTimeMs + bonusTime,
      );
    }
  }

  int _calculateScore(int charCount, int combo) {
    final baseScore = charCount * 10;
    final comboBonus = combo * 2;
    final multiplier = switch (state.difficulty) {
      'beginner' => 1.0,
      'intermediate' => 1.5,
      'advanced' => 2.0,
      _ => 1.0,
    };
    return ((baseScore + comboBonus) * multiplier).round();
  }

  int _calculateCharacterLevel(int score) {
    const thresholds = [0, 500, 1500, 3000, 5000, 10000];
    for (int i = thresholds.length - 1; i >= 0; i--) {
      if (score >= thresholds[i]) return i;
    }
    return 0;
  }

  void _endGame() {
    _gameTimer?.cancel();
    state = state.copyWith(
      isPlaying: false,
      isFinished: true,
    );
  }

  void deleteLastCharacter() {
    if (state.inputBuffer.isEmpty) return;
    state = state.copyWith(
      inputBuffer: state.inputBuffer.substring(0, state.inputBuffer.length - 1),
    );
  }
}
